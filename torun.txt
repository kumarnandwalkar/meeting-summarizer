docker network create meeting-summarizer-net

docker volume create meeting-summarizer-mongo

docker run -d --name meeting-summarizer-mongo --network meeting-summarizer-net -p 27017:27017 -v meeting-summarizer-mongo:/data/db mongo:7

docker build -t meeting-summarizer:latest -f meeting-summarizer/Dockerfile meeting-summarizer

docker run -d --name meeting-summarizer-app \
  --network meeting-summarizer-net \
  -p 5173:5173 \
  -p 8080:8080 \
  -e PORT=8080 \
  -e MONGODB_URI=mongodb://meeting-summarizer-mongo:27017/meeting_summarizer \
  -e CORS_ORIGIN=http://YOUR_PUBLIC_HOST_OR_IP \
  -e AI_PROVIDER=groq \
  -e GROQ_API_KEY=YOUR_GROQ_API_KEY \
  -e EMAIL_PROVIDER=resend \
  -e RESEND_API_KEY=YOUR_RESEND_API_KEY \
  -e MAIL_FROM="Team <no-reply@example.com>" \
  -e VITE_API_BASE=http://YOUR_PUBLIC_HOST_OR_IP:8080/api \
  meeting-summarizer:latest
